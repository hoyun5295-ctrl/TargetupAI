```bash

# (값은 노출하지 않고 키 이름만 확인)
cd /home/administrator/targetup-app/packages/backend
grep -oP '^[A-Z_]+(?==)' .env | sort
```

```bash
# JWT_SECRET이 기본값인지 확인 (보안상 값은 안 봐도 됨, 설정 여부만)
grep -c 'JWT_SECRET=' .env
# 결과가 1이면 설정됨, 0이면 미설정
```

```bash
# 핵심 환경변수 설정 여부 확인
for key in JWT_SECRET DATABASE_URL MYSQL_PASSWORD ANTHROPIC_API_KEY OPT_OUT_080_TOKEN SMS_TABLES; do
  if grep -q "^${key}=" .env; then
    echo "✅ $key = 설정됨"
  else
    echo "❌ $key = 미설정"
  fi
done
```
